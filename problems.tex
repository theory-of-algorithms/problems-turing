  Then the number on the tape is $8+4+1=13$.

        \midrule
            $q_0$ & $\sqcup$ & 0 & R & $q_a$ \\
            $q_0$ & 0 & 0 & R & $q_0$ \\
            $q_0$ & 1 & 1 & R & $q_0$ \\
        \bottomrule
        \hline
      \end{tabular}
    \end{table}
  \end{solution}
